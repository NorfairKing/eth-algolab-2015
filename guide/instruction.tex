\section{Models}
\subsection{Greedy}
\begin{itemize}
  \item Make sure any local optimum is a global optimum.
  \item Define the 'best' guess.
\end{itemize}

\subsection{Divide and conquer}
\begin{itemize}
  \item Define how to divide.
  \item Define how to merge results.
\end{itemize}

\subsection{DFS}
\begin{itemize}
  \item Define nodes.
  \item Define edges.
\end{itemize}

\subsection{BFS}
\begin{itemize}
  \item Define nodes.
  \item Define edges.
\end{itemize}

\subsection{Minimum Spanning Tree}
\begin{itemize}
  \item Undirected graph.
  \item Define nodes.
  \item Define edges.
  \item Define edge costs.
\end{itemize}

\subsection{Graph coloring}
\begin{itemize}
  \item Define nodes.
  \item Define edges.
  \item Define colors.
\end{itemize}

\subsection{Graph matching}
\begin{itemize}
  \item Define nodes.
  \item Define edges.
  \item A matching in a graph is a set of edges without common vertices.
\end{itemize}

\subsection{Topological sort}
\begin{itemize}
  \item Directed graph.
  \item Define nodes.
  \item Define edges.
\end{itemize}

\subsection{Linear programming}
\begin{itemize}
  \item Define the objective function to minimize/maximize. Both the factors and the variables.
  \item Define the constraints.
\end{itemize}

\subsection{Network flow: Max flow}
\begin{itemize}
  \item Define nodes
  \item Define edges.
  \item Define the source and the sink.
  \item Define flow capacity for each edge
\end{itemize}

\subsection{Network flow: Max flow/Min cost}
\begin{itemize}
  \item Define nodes
  \item Define edges.
  \item Define the source and the sink.
  \item Define flow capacity for each edge.
  \item Define flow cost for each edge.
\end{itemize}
